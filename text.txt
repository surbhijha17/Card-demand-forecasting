import pandas as pd

# Sample DataFrame
data = {'A': [1000.123456789, -2000.987654321],
        'B': [123456, -987654]}
df = pd.DataFrame(data)

# Define custom formatting function
def custom_format(x):
    if x > 0:
        return '{:,.2f}'.format(x)
    elif x < 0:
        return '({:,.2f})'.format(abs(x))
    else:
        return '-'

# Apply custom formatting to all columns
styled_df = df.style.format(custom_format)

# Display the styled DataFrame
styled_df
