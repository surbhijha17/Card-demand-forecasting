from pptx import Presentation

def get_cell_positions(presentation_path, slide_index, table_index, row_index, col_index):
    prs = Presentation(presentation_path)
    slide = prs.slides[slide_index]
    table = slide.shapes[table_index].table

    # Get the cell
    cell = table.cell(row_index, col_index)

    # Get the position of the cell
    left = cell._element.left
    top = cell._element.top
    width = cell._element.width
    height = cell._element.height

    # Calculate start and end positions
    start_position = (left, top)
    end_position = (left + width, top + height)

    return start_position, end_position

# Example usage
presentation_path = 'example.pptx'
slide_index = 0
table_index = 0
row_index = 0
col_index = 0

start_position, end_position = get_cell_positions(presentation_path, slide_index, table_index, row_index, col_index)
print("Start Position:", start_position)
print("End Position:", end_position)
