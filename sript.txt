from flask import Flask, render_template, request, jsonify
from werkzeug.utils import secure_filename
import os

app = Flask(__name__)
app.config['UPLOAD_FOLDER'] = 'static/uploads'  # Folder to store uploaded files
app.config['ALLOWED_EXTENSIONS'] = {'mp3', 'wav'}  # Allowed file extensions

def allowed_file(filename):
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in app.config['ALLOWED_EXTENSIONS']

@app.route('/', methods=['GET', 'POST'])
def home():
    if request.method == 'POST' and 'audio_file' in request.files:
        file = request.files['audio_file']
        if file and allowed_file(file.filename):
            filename = secure_filename(file.filename)
            file.save(os.path.join(app.config['UPLOAD_FOLDER'], filename))
            return jsonify({'file_path': f'static/uploads/{filename}',
                            'annotations': [
                                {'timestamp': 1, 'label': 'Annotation 1'},
                                {'timestamp': 8, 'label': 'Annotation 2'},
                                {'timestamp': 10, 'label': 'Annotation 3'}
                            ]})

    return render_template('index.html')

def main():
    app.run(debug=True)

if __name__ == '__main__':
    main()
