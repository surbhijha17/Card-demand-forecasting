<!DOCTYPE html>
<html>
<head>
    <title>AJAX POST Request Example</title>
    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
    <script>
        $(document).ready(function() {
            $('#submitBtn').click(function(e) {
                e.preventDefault();
                
                var dictionaryData = $('#dictionaryInput').val();
                var stringData = $('#stringInput').val();

                var requestData = {
                    dictionaryData: dictionaryData,
                    stringData: stringData
                };

                var jsonData = JSON.stringify(requestData);

                $.ajax({
                    url: '/process_data',
                    type: 'POST',
                    contentType: 'application/json',
                    data: jsonData,
                    success: function(response) {
                        console.log(response.message);
                    }
                });
            });
        });
    </script>
</head>
<body>
    <h1>AJAX POST Request Example</h1>
    <form>
        <label for="dictionaryInput">Dictionary Data:</label>
        <input type="text" id="dictionaryInput" name="dictionaryData"><br><br>

        <label for="stringInput">String Data:</label>
        <input type="text" id="stringInput" name="stringData"><br><br>

        <button id="submitBtn" type="submit">Submit</button>
    </form>
</body>
</html>
