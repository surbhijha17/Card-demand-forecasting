import streamlit as st
import json

def main():
    st.title("Editable Dictionary Viewer and Updater")

    # Initialize the dictionary
    my_dictionary = {"key1": "value1", "key2": "value2", "key3": "value3"}

    # Convert dictionary to JSON string
    json_string = json.dumps(my_dictionary, indent=4)

    st.write("### Current Dictionary Contents:")
    st.text_area("Edit Dictionary", value=json_string, height=300)

    # Add a section for updating the dictionary
    st.write("### Update Dictionary")
    new_json_string = st.text_area("Enter Updated JSON", height=100)
    if st.button("Update"):
        try:
            updated_dictionary = json.loads(new_json_string)
            my_dictionary.update(updated_dictionary)
            st.write("### Dictionary Updated!")
            st.write(my_dictionary)
        except json.JSONDecodeError:
            st.error("Invalid JSON format. Please enter a valid JSON.")

if __name__ == "__main__":
    main()
