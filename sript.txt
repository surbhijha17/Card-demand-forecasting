import streamlit as st

def display_dictionary(dictionary):
    st.write("### Dictionary Contents:")
    for key, value in dictionary.items():
        st.write(f"- **{key}:** {value}")

def update_dictionary(dictionary, key, value):
    dictionary[key] = value
    st.write("### Dictionary Updated!")
    st.write(f"- **{key}:** {value}")

def main():
    st.title("Dictionary Viewer and Updater")

    # Initialize the dictionary
    my_dictionary = {"key1": "value1", "key2": "value2", "key3": "value3"}

    # Display the initial contents of the dictionary
    display_dictionary(my_dictionary)

    # Add a section for updating the dictionary
    st.sidebar.title("Update Dictionary")
    key = st.sidebar.text_input("Enter Key:")
    value = st.sidebar.text_input("Enter Value:")
    if st.sidebar.button("Update"):
        update_dictionary(my_dictionary, key, value)
        # Display the updated contents of the dictionary
        display_dictionary(my_dictionary)

if __name__ == "__main__":
    main()
